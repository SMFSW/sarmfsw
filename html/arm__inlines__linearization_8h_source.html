47__INLINE SDWORD NONNULL__ linearization_eval(const SDWORD array[][2], const DWORD nb, const SDWORD val)
48{
49 DWORD idxMin = 0, idxMax = nb - 1U;
50
51 // Find indexes
52 while (idxMax > (idxMin + 1U))
53 {
54 const DWORD idx = (((idxMax - idxMin) / 2U) + idxMin);
55
56 if (val < array[idx][0]) { idxMax = idx; }
57 else { idxMin = idx; }
58 }
59
60 // Get interval values
61 const SDWORD x0 = array[idxMin][0];
62 const SDWORD y0 = array[idxMin][1];
63 const SDWORD x1 = array[idxMax][0];
64 const SDWORD y1 = array[idxMax][1];
65
66 // Linearize
67 const float coef = (float) (y1 - y0) / (float) (x1 - x0);
68 return (SDWORD) (coef * (val - x0) + y0);
69}
