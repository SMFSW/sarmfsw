127{
128 const intCPU_t diff_bits = MIN(32U, to) - MIN(32U, from);
129 const uintCPU_t shift = abs(diff_bits);
130 DWORD conv = val & maskBits(from);

132 if (diff_bits > 0) { conv = LSHIFT(val, shift) | RSHIFT(val, from - shift); }
